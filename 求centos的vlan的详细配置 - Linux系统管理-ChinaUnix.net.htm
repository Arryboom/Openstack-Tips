#!/bin/bash

/sbin/modprobe 8021q

/sbin/vconfig add eth1 1
/sbin/vconfig add eth1 2
/sbin/vconfig add eth1 3
/sbin/vconfig add eth1 4

/sbin/ip address add 192.168.2.1/24 dev eth1.1
/sbin/ip address add 192.168.3.1/24 dev eth1.2
/sbin/ip address add 192.168.4.1/24 dev eth1.3
/sbin/ip address add 192.168.5.1/24 dev eth1.4

/sbin/ip link set eth1.1 up
/sbin/ip link set eth1.2 up
/sbin/ip link set eth1.3 up
/sbin/ip link set eth1.4 up
